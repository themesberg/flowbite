The phone number input component from Flowbite can be used to set a phone number inside a form field by using the native `type="tel"` attribute and also use a dropdown menu to select the country code.

The examples are built with the utility classes from Tailwind CSS and they are fully responsive, dark mode compatible and support RTL layouts and can be used for any type of web project.

## Default phone input

Use this component to set a phone number inside an input field by setting the `type="tel"` attribute.

<script>
    document.addEventListener('DOMContentLoaded', function() {
        const defaultFlag = document.querySelector('.flag-option[data-flag="+1"]');
        if (defaultFlag) {
            const selectedFlag = defaultFlag.getAttribute('data-flag');
            const selectedSvg = defaultFlag.querySelector('svg').cloneNode(true);
            const flagContainer = document.getElementById('selected-flag');

            flagContainer.innerHTML = '';
            flagContainer.appendChild(selectedSvg);
            updateButtonLabel(selectedFlag);
        }
    });

    document.getElementById('dropdown-phone-button').addEventListener('click', function() {
        document.getElementById('dropdown-phone').classList.toggle('hidden');
    });

    document.querySelectorAll('.flag-option').forEach(function(button) {
        button.addEventListener('click', function() {
            const selectedFlag = this.getAttribute('data-flag');
            const selectedSvg = this.querySelector('svg').cloneNode(true);
            const flagContainer = document.getElementById('selected-flag');

            flagContainer.innerHTML = '';
            flagContainer.appendChild(selectedSvg);
            updateButtonLabel(selectedFlag);
            document.getElementById('dropdown-phone').classList.add('hidden');
        });
    });

    function updateButtonLabel(selectedFlag) {
        const button = document.getElementById('dropdown-phone-button');
        button.innerHTML = `
            <svg id="selected-flag" fill="none" aria-hidden="true" class="h-4 w-4 me-2" viewBox="0 0 20 15"></svg>
            ${selectedFlag} <svg class="w-2.5 h-2.5 ms-2.5" aria-hidden="true" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 10 6"><path stroke="currentColor" stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="m1 1 4 4 4-4"/></svg>
        `;
        const selectedSvg = document.querySelector(`.flag-option[data-flag="${selectedFlag}"] svg`).cloneNode(true);
        const flagContainer = button.querySelector('#selected-flag');
        flagContainer.innerHTML = '';
        flagContainer.appendChild(selectedSvg);
    }
</script>
